# `@rainbow-me/kit-ui`

Collection of RainbowKit UI components.

## Profile

User badge for your dapp. It displays the users wallet address, their balance, and what method they are connected with (e.g. rainbow, walletconnect, metamask, coinbase, etc).
Use it to indicate that the user's wallet is connected by replacing your "Connect Wallet" open modal button.

```jsx
<Profile
  rpcProvider={mainnetProvider} // explorer provider
  copyAddress
  ipfsGatewayUrl="ipfs.io" // IPFS gateway for avatars starting with `ipfs://`
  modalOptions={{ // same options as useModal
    chains: ['mainnet', 'polygon', 'optimism', 'arbitrum', 'kovan', 'ropsten'],
    wallets: [
      'metamask',
      'coinbase',
      {
        name: 'rainbow',
        connectorName: 'walletconnect',
        options: {
          rpc: walletConnectRPCs
        } as WalletConnectConnectorArguments
      }
    ],
    terms: ( // terms of usage shown in the below of a modal
      <>
        By connecting, you acknowledge that youâ€™ve read and agree to the{' '}
        <a>RainbowKit&apos;s Terms of Service.</a>
      </>
    )
  }}
/>
```

const EthAddressPicExample = () => {

  return <EthAddress addr="foda.eth" profileIcon={data.records?.avatar as string} />
}

Get explorer based on network ID:

```tsx
import React, { useState, useEffect } from 'react'
import { Tx } from '@rainbow-me/kit-ui'

const TxView = () => {
  const { library: provider } = useWeb3React()
  const [chainId, set] = useState(1)

  useEffect(() => {
    provider?.getNetwork(({ chainId }) => set(chainId))
  }, [provider])

  return (
    <Tx
      chainId={chainId}
      status="success"
      title="Approve USDT for trade on 1INCH V3"
      chainId={1}
      hash="0x4d7a50148d6aefff1a63571818d8072f4cccdd331f77699b77d314847c2d3058"
    />
  )
}
```

## TxHistory

```tsx
import { TxHistory } from '@rainbow-me/kit-ui'
import { useExplorerTxHistory, useWeb3State } from '@rainbow-me/kit-hooks'
import { ChainProvider } from 'chain-provider'

const { address, chainId, provider } = useWeb3State()
const [latestBlock, setLatestBlock] = useState(12862901)

useConnectOnMount(injected, true)

const { data: txes } = useExplorerTxHistory({
  fetcher: etherscanFetcher,
  provider: new ChainProvider('mainnet'),
  address,
  options: {
    fromBlock: latestBlock
  }
})

const App = () => {
  useEffect(() => {
    if (provider) provider.getBlockNumber().then((block) => setLatestBlock(block - 100_000))
  }, [provider])

  return <TxHistory {...{ chainId, txes }} />
}
```
